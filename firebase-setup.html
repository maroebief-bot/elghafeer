1
إنشاء مشروع Firebase
روح console.firebase.google.com واعمل الخطوات دي:
  • اضغط Add Project → اكتب اسم المشروع: elghafeer
  • Google Analytics → اضغط Continue (Enable أو Disable مش مهم)
  • اضغط Create Project → انتظر التحميل
  • من الـ sidebar اضغط على Project Settings ⚙️
  • اختار General → تحت "Your apps" اضغط </> Web
  • اكتب اسم التطبيق: elghafeer-web → Register app
  • انسخ الـ firebaseConfig وحطها في ملف firebase-config.js
2
تفعيل Authentication (OTP)
  • من الـ sidebar: Build → Authentication → Get Started
  • اضغط Sign-in method
  • اختار Phone → اضغط Enable → Save
  • اضغط Email/Password → Enable → Save (للأدمن)
✅ OTP هيشتغل على أرقام مصرية (+20) بعد التفعيل مباشرة
⚠️ للتجربة على أجهزة حقيقية محتاج تضيف رقمك في قسم "Phone numbers for testing" أو تستخدم رقم فعلي
3
إنشاء Firestore Database
  • من الـ sidebar: Build → Firestore Database → Create Database
  • اختار Start in test mode → Next
  • اختار eur3 (Europe) أو me-central1 (الأقرب) → Enable

5
Firestore Security Rules
روح Firestore Database → Rules → امسح كل حاجة موجودة والصق ده:
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isAuth() {
      return request.auth != null;
    }
    function getAdmin() {
      return get(/databases/$(database)/documents/admins/$(request.auth.uid));
    }
    function getRole() {
      return getAdmin().data.role;
    }
    function isSuperAdmin() {
      return isAuth() && getRole() == 'superadmin';
    }
    function isRegionManager() {
      return isAuth() && getRole() in ['superadmin', 'region_manager'];
    }
    function isSupervisor() {
      return isAuth() && getRole() in ['superadmin', 'region_manager', 'supervisor'];
    }
    function isStaff() {
      return isAuth() && getRole() in ['superadmin', 'region_manager', 'supervisor', 'street_monitor'];
    }
    function isOwner(uid) {
      return isAuth() && request.auth.uid == uid;
    }

    // ADMINS
    match /admins/{uid} {
      allow read: if isStaff();
      allow create, update: if isSuperAdmin();
      allow delete: if isSuperAdmin();
    }

    // DRIVERS
    match /drivers/{uid} {
      allow read: if isAuth();
      allow create: if isAuth() && isOwner(uid);
      allow update: if isOwner(uid) || isSupervisor();
      allow delete: if isSuperAdmin();
    }

    // CLIENTS
    match /clients/{uid} {
      allow read: if isOwner(uid) || isSupervisor();
      allow create: if isAuth() && isOwner(uid);
      allow update: if isOwner(uid) || isRegionManager();
      allow delete: if isSuperAdmin();
    }

    // RIDES
    match /rides/{rideId} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isAuth();
      allow delete: if isSuperAdmin();
    }

    // SOS
    match /sos/{id} {
      allow read: if isSupervisor();
      allow create: if isAuth();
      allow update: if isSupervisor();
      allow delete: if isRegionManager();
    }

    // FAMILIES
    match /families/{id} {
      allow read, write: if isAuth();
    }

    // PARTNERS (workshops, parts shops, companies)
    match /partners/{id} {
      allow read: if isAuth();
      allow create: if isAuth();
      allow update: if isAuth() && (
        resource.data.ownerId == request.auth.uid || isSupervisor()
      );
      allow delete: if isRegionManager();
    }

    // PAYMENTS
    match /payments/{id} {
      allow read: if isAuth() && (
        resource.data.userId == request.auth.uid || isSupervisor()
      );
      allow create: if isAuth();
      allow update: if isRegionManager();
      allow delete: if isSuperAdmin();
    }

    // ERP EMPLOYEES
    match /erp_employees/{id} {
      allow read: if isRegionManager();
      allow write: if isSuperAdmin();
    }

    // ERP TRANSACTIONS
    match /erp_transactions/{id} {
      allow read: if isRegionManager();
      allow write: if isRegionManager();
    }

    // NOTIFICATIONS
    match /notifications/{id} {
      allow read: if isAuth() && (
        resource.data.targetId == request.auth.uid || isSupervisor()
      );
      allow create: if isSupervisor();
      allow update: if isAuth();
      allow delete: if isSupervisor();
    }

    // REGIONS
    match /regions/{id} {
      allow read: if isAuth();
      allow write: if isSuperAdmin();
    }

    // DISCOUNT CONFIGS (for partners)
    match /discount_configs/{id} {
      allow read: if isAuth();
      allow write: if isSuperAdmin();
    }

    // DRIVER PAYMENT CONFIG
    match /driver_payment_configs/{uid} {
      allow read: if isOwner(uid) || isSupervisor();
      allow write: if isSuperAdmin();
    }
  }
}
✅ بعد اللصق اضغط Publish — الـ Rules هتتفعل فوراً
6
Storage Rules
روح Storage → Rules والصق ده:
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    match /drivers/{uid}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == uid;
    }
    match /partners/{uid}/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && request.auth.uid == uid;
    }
    match /sos_recordings/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null;
    }
  }
}
